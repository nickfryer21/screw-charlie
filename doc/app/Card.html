def has_valid_suit?

  unless [:Hearts, :Clubs, :Spades, :Diamonds].include? self.suit.to_sym

    self.errors.add(:base, "Never heard of that suit")

  end

end

def rank_in_range?

def to_s

  case self.rank

    when 1 then

      'Ace'

    when 11 then

      'Jack'

    when 12 then

      'Queen'

    when 13 then

      'King'

    else

      self.rank.to_s

  end + ' of ' + self.suit.to_s

end
